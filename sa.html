class DeviceRecognizer(private val context: Context) { private val rgbMat = Mat() private val grayMat = Mat() private val edges = Mat() fun detectDeviceContour(bitmap: Bitmap): List { // 将Bitmap转换为Mat Utils bitmapToMat(bitmap, rgbMat) // 转换为灰度图 Imgproc cvtColor(rgbMat, grayMat, Imgproc COLOR_RGB2GRAY) // 高斯模糊降噪 Imgproc GaussianBlur(grayMat, grayMat, Size(5 0, 5 0), 0 0) // Canny边缘检测 Imgproc Canny(grayMat, edges, 50 0, 150 0) // 查找轮廓 val contours = ArrayList() val hierarchy = Mat() Imgproc findContours(edges, contours, hierarchy, Imgproc RETR_EXTERNAL, Imgproc CHAIN_APPROX_SIMPLE) // 找到最大轮廓（假设是设备） val largestContour = contours maxByOrNull { it total() } ?: return emptyList() // 转换为点列表 return largestContour toList() map { Point(it x toFloat(), it y toFloat()) } } }
