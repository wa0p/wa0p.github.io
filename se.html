// 定义气泡可能的颜色范围（透明区域在HSV中表现为低饱和度） val lowerThreshold = Scalar(0, 0, 50) val upperThreshold = Scalar(180, 30, 255) // 创建掩码 val mask = Mat() Core inRange(hsv, lowerThreshold, upperThreshold, mask) // 查找轮廓 val contours = ArrayList() val hierarchy = Mat() Imgproc findContours(mask, contours, hierarchy, Imgproc RETR_EXTERNAL, Imgproc CHAIN_APPROX_SIMPLE) // 过滤小轮廓并转换为矩形 for (contour in contours) { val rect = Imgproc boundingRect(contour) if (rect width > 20 && rect height > 20) { // 过滤过小的区域 result add(rect) } } return result } }
